# 🔌 插件开发指南

## 概述

OpenClaw 支持插件扩展，可以添加新的渠道、功能和集成。

## 插件结构

```
my-plugin/
├── package.json
├── index.js
└── README.md
```

## 基本示例

```javascript
// index.js
export default {
  name: 'my-plugin',
  version: '1.0.0',
  description: '我的第一个插件',
  
  // 插件初始化
  async onLoad(config) {
    console.log('插件加载:', this.name);
    this.config = config;
  },
  
  // 消息处理
  async onMessage(message) {
    // 处理收到的消息
    return message;
  },
  
  // 消息发送前
  async onSend(message) {
    // 发送前处理
    return message;
  },
  
  // 定时任务
  schedule: '0 * * * *', // 每小时
  
  async onScheduled() {
    console.log('定时任务执行');
  },
  
  // 插件卸载
  async onUnload() {
    console.log('插件卸载');
  }
}
```

## 高级功能

### 自定义命令

```javascript
export default {
  name: 'command-plugin',
  
  commands: [
    {
      name: 'hello',
      description: '打招呼',
      handler: async (args, message) => {
        return '你好！我是插件机器人 🦊';
      }
    },
    {
      name: 'echo',
      description: '回声',
      options: [
        { name: 'text', type: 'string', required: true }
      ],
      handler: async (args, message) => {
        return args.text;
      }
    }
  ]
}
```

### 自定义 API

```javascript
export default {
  name: 'api-plugin',
  
  // 注册 API 端点
  api: [
    {
      method: 'GET',
      path: '/my-plugin/status',
      handler: async (req, res) => {
        return { status: 'ok', uptime: process.uptime() };
      }
    },
    {
      method: 'POST',
      path: '/my-plugin/webhook',
      handler: async (req, res) => {
        const { event, data } = req.body;
        // 处理 webhook
        return { received: true };
      }
    }
  ]
}
```

### 工具注册

```javascript
export default {

          param1: { type: 'string' }
        },
        required: ['param1']
      },
      handler: async (params) => {
        return { result: params.param1 };
      }
    }
  ]
}
```

## 完整示例：天气插件

```javascript
import axios from 'axios';

export default {
  name: 'weather-plugin',
  version: '1.0.0',
  description: '天气查询插件',
  
  config: {
    apiKey: '',
    defaultCity: '北京'
  },
  
  // 默认配置
  defaults: {
    apiKey: '',
    defaultCity: '北京'
  },
  
  async onLoad(config) {
    this.config = { ...this.defaults, ...config };
    
    // 验证配置
    if (!this.config.apiKey) {
      console.warn('天气插件：未配置 API Key');
    }
  },
  
  rules: [
    {
      // 匹配天气查询
      pattern: /^天气\s*(.+)?/,
      handler: async (match) => {
        const city = match[1] || this.config.defaultCity;
        return await this.getWeather(city);
      }
    }
  ],
  
  async getWeather(city) {
    if (!this.config.apiKey) {
      return '请先配置天气 API Key';
    }
    
    try {
      // 调用天气 API（示例）
      const data = {
        city,
        temp: 22,
        condition: '晴',
        humidity: 65
      };
      
      return `🌤️ ${city}天气

温度：${data.temp}°C
天气：${data.condition}
湿度：${data.humidity}%`;
    } catch (error) {
      return '查询失败，请稍后重试';
    }
  },
  
  // 定时推送
  schedule: '0 7 * * *',
  
  async onScheduled() {
    const cities = ['北京', '上海', '广州'];
    const channel = 'telegram';
    
    for (const city of cities) {
      const weather = await this.getWeather(city);
      await this.sendToChannel(channel, {
        target: 'weather-channel',
        message: weather
      });
    }
  },
  
  // 插件信息
  info() {
    return {
      name: this.name,
      version: this.version,
      status: this.config.apiKey ? 'ready' : 'unconfigured'
    };
  }
}
```

## 发布

1. 创建 GitHub 仓库
2. 添加 `package.json`
3. 添加 `README.md`
4. 发布到 npm（可选）

```bash
# 本地测试
cp -r my-plugin ~/.openclaw/workspace/skills/
openclaw skills enable my-plugin
```

## 常用 API

| 方法 | 说明 |
|------|------|
| `this.sendToChannel(channel, msg)` | 发送消息 |
| `this.sendMessage(user, msg)` | 发送消息给用户 |
| `this.getUserPref(userId)` | 获取用户偏好 |
| `this.setUserPref(userId, key, value)` | 设置用户偏好 |
| `this.delegateTo(agent, msg)` | 转发到其他代理 |
| `this.scheduleTask(task)` | 创建定时任务 |
