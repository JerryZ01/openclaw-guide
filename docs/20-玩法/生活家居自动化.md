## 1. 智能家居控制

### 功能
- 语音控制灯、空调、门锁
- 场景联动（回家/离家/睡觉）
- 语音播报天气/日程

### 支持设备
- 米家 (Xiaomi Home)
- Home Assistant
- Apple HomeKit
- Aqara
- IKEA Tradfri

### 配置

```json
{
  "skills": {
    "smart-home": {
      "enabled": true,
      "platform": "homeassistant",
      "url": "http://homeassistant.local:8123",
      "token": "YOUR_LONG_LIVED_TOKEN",
      
      "devices": {
        "客厅灯": { "entity": "light.living_room", "type": "light" },
        "卧室空调": { "entity": "climate.bedroom", "type": "climate" },
        "大门锁": { "entity": "lock.front_door", "type": "lock" },
        "窗帘": { "entity": "cover.curtain", "type": "cover" }
      },
      
      "scenes": {
        "回家": ["开灯", "开空调", "开门锁"],
        "离家": ["关灯", "关空调", "门锁"],
        "睡觉": ["关灯", "窗帘关闭", "门锁"]
      }
    }
  }
}
```

### 实现

```javascript
// skills/smart-home/index.js
module.exports = {
  name: "smart-home",
  version: "1.0.0",
  
  rules: [
    {
      pattern: /(开|关|打开|关闭|调(.+)亮度)(.+)/,
      handler: async (match, context) => {
        const action = match[1].includes('开') ? 'on' : 'off';
        const device = match[3].trim();
        
        await this.homeAssistant.toggle(device, action);
        return `✅ 已${action === 'on' ? '开启' : '关闭'} ${device}`;
      }
    },
    {
      pattern: /执行\s*(.+)/,
      handler: async (match) => {
        const scene = match[1];
        const sceneConfig = this.config.scenes[scene];
        
        for (const action of sceneConfig) {
          await this.executeAction(action);
        }
        
        return `✅ 已执行场景：${scene}`;
      }
    },
    {
      pattern: /温度调到(.+)/,
      handler: async (match) => {
        const temp = match[1];
        await this.homeAssistant.setClimate('bedroom', temp);
        return `✅ 空调温度已设置为 ${temp} 度`;
      }
    }
  ],
  
  async executeAction(action) {
    const [cmd, device] = action.split(' ');
    await this.homeAssistant.toggle(device, cmd === '开' ? 'on' : 'off');
  }
};
```

## 2. 早晨简报 (Daily Brief)

> 最受欢迎的玩法之一！每天早上自动推送

### 功能
- 天气 + 日程 + 待办 + 新闻
- 可定制内容
- 语音播报（可选）

### 配置

```json
{
  "skills": {
    "daily-brief": {
      "enabled": true,
      "schedule": "30 7 * * 1-5", // 工作日 7:30
      "channels": ["feishu", "telegram"],
      "sections": {
        "weather": { "city": "上海", "enabled": true },
        "calendar": { "enabled": true },
        "tasks": { "enabled": true, "source": "notion" },
        "news": { "enabled": true, "keywords": ["AI", "科技"] },
        "joke": { "enabled": true }
      }
    }
  }
}
```

### 实现

```javascript
// skills/daily-brief/index.js
module.exports = {
  name: "daily-brief",
  version: "1.0.0",
  
  schedule: "30 7 * * 1-5",
  
  async onScheduled() {
    const sections = [];
    
    // 1. 天气
    const weather = await this.getWeather('上海');
    sections.push(this.formatWeather(weather));
    
    // 2. 日程
    const events = await this.getTodayEvents();
    sections.push(this.formatEvents(events));
    
    // 3. 待办
    const tasks = await this.getPendingTasks();
    sections.push(this.formatTasks(tasks));
    
    // 4. 新闻
    const news = await this.getNews(['AI', '科技']);
    sections.push(this.formatNews(news));
    
    // 5. 每日一言
    const quote = await this.getQuote();
    sections.push(this.formatQuote(quote));
    
    // 发送简报
    const brief = this.assembleBrief(sections);
    await this.broadcast(brief);
  },
  
  assembleBrief(sections) {
    return `☀️ **早上好！**\n\n${sections.join('\n\n')}\n\n💡 美好的一天从此刻开始！`;
  },
  
  formatWeather(w) {
    return `**今日天气** ${w.temp}°C ${w.condition}\n${w.suggestion}`;
  },
  
  formatEvents(events) {
    if (!events.length) return '**📅 今日日程**：无';
    return '**📅 今日日程**\n' + events.map(e => 
      `- ${e.time} ${e.title}`
    ).join('\n');
  },
  
  formatTasks(tasks) {
    if (!tasks.length) return '**✅ 待办事项**：无';
    return '**✅ 待办事项**\n' + tasks.map(t => 
      `- ${t.priority === 'high' ? '🔴' : '🟡'} ${t.title}`
    ).join('\n');
  }
};
```

## 3. 包裹物流追踪

### 功能
- 从邮件/短信提取快递单号
- 自动查物流状态
- 状态变更通知
- 简单仪表盘

### 配置

```json
{
  "skills": {
    "package-tracker": {
      "enabled": true,
      "schedule": "0 8-20/4 * * *", // 每4小时检查
      "sources": ["gmail", "sms"],
      "carriers": ["顺丰", "中通", "圆通", "京东", "菜鸟"],
      "notifyOnStatus": ["派送中", "已签收", "异常"]
    }
  }
}
```

### 实现

```javascript
// skills/package-tracker/index.js
module.exports = {
  name: "package-tracker",
  version: "1.0.0",
  
  // 正则匹配快递单号
  trackingPatterns: [
    /[A-Z]{2}\d{9,13}/,  // 国际
    /\d{12,15}/,         // 国内
  ],
  
  async onScheduled() {
    // 1. 扫描新邮件/短信获取快递单号
    const newNumbers = await this.scanForTrackingNumbers();
    
    // 2. 添加到追踪列表
    for (const num of newNumbers) {
      await this.addToTracker(num);
    }
    
    // 3. 检查所有快递状态
    const packages = await this.getAllPackages();
    
    for (const pkg of packages) {
      const status = await this.checkStatus(pkg.carrier, pkg.number);
      
      if (status !== pkg.lastStatus) {
        // 状态变化，通知用户
        await this.notifyUser(pkg, status);
        await this.updateStatus(pkg.id, status);
      }
    }
  },
  
  async scanForTrackingNumbers() {
    const emails = await this.gmail.search('subject:快递 OR subject:发货');
    const numbers = [];
    
    for (const email of emails) {
      const matches = this.trackingPatterns.map(p => 
        email.body.match(p)
      ).filter(Boolean);
      numbers.push(...matches);
    }
    
    return [...new Set(numbers)];
  }
};
```
