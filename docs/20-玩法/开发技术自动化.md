## 1. 手机写代码

> 在 Telegram/WhatsApp 聊天中下指令就能部署代码！

### 功能
- 通过聊天指令执行 git 操作
- 部署服务
- 查看日志
- 重启服务

### 配置

```json
{
  "skills": {
    "mobile-dev": {
      "enabled": true,
      "allowedCommands": [
        "git pull",
        "npm run",
        "docker",
        "systemctl",
        "tail",
        "ps"
      ],
      "repos": [
        {
          "name": "my-app",
          "path": "/var/www/my-app",
          "branch": "main"
        }
      ]
    }
  }
}
```

### 实现

```javascript
// skills/mobile-dev/index.js
module.exports = {
  name: "mobile-dev",
  version: "1.0.0",
  
  rules: [
    {
      pattern: /^git\s+pull/,
      handler: async (match, context) => {
        const { repo, branch } = this.parseRepo(match[0]);
        const result = await this.exec(`cd ${repo.path} && git pull origin ${branch}`);
        return `\`\`\`\n${result}\n\`\`\``;
      }
    },
    {
      pattern: /^npm\s+(.+)/,
      handler: async (match) => {
        const cmd = match[1];
        const result = await this.exec(`npm ${cmd}`);
        return `\`\`\`\n${result}\n\`\`\`;
      }
    },
    {
      pattern: /^docker\s+(.+)/,
      handler: async (match) => {
        const cmd = match[1];
        const result = await this.exec(`sudo docker ${cmd}`);
        return `\`\`\`\n${result}\n\`\`\``;
      }
    },
    {
      pattern: /^restart\s+(.+)/,
      handler: async (match) => {
        const service = match[1];
        await this.exec(`sudo systemctl restart ${service}`);
        return `✅ 已重启 ${service}`;
      }
    },
    {
      pattern: /^logs\s+(.+)/,
      handler: async (match) => {
        const service = match[1];
        const lines = match[2] || 50;
        const result = await this.exec(`sudo journalctl -u ${service} -n ${lines} --no-pager`);
        return `\`\`\`\n${result}\n\`\`\``;
      }
    },
    {
      pattern: /^status/,
      handler: async () => {
        const services = ['nginx', 'docker', 'openclaw'];
        let status = '🖥️ 服务状态\n\n';
        
        for (const svc of services) {
          const result = await this.exec(`systemctl is-active ${svc}`);
          const emoji = result.trim() === 'active' ? '✅' : '❌';
          status += `${emoji} ${svc}\n`;
        }
        
        return status;
      }
    }
  ],
  
  parseRepo(cmd) {
    // 解析命令，提取仓库和分支
    return { repo: this.config.repos[0], branch: 'main' };
  }
};
```

## 2. 自愈服务器

### 功能
- 自动监控服务器状态
- 检测服务异常
- 自动重启服务
- 告警通知

### 配置

```json
{
  "skills": {
    "self-healing": {
      "enabled": true,
      "schedule": "*/5 * * * *", // 每5分钟检查
      "checks": {
        "cpu": { "threshold": 90, "duration": 300 },
        "memory": { "threshold": 90, "duration": 300 },
        "disk": { "threshold": 85, "duration": 0 },
        "services": ["nginx", "docker", "mysql"]
      },
      "actions": {
        "high_cpu": ["restart-heavy-process", "notify"],
        "service_down": ["restart-service", "notify"],
        "disk_full": ["cleanup-logs", "notify"]
      }
    }
  }
}
```

### 实现

```javascript
// skills/self-healing/index.js
module.exports = {
  name: "self-healing",
  version: "1.0.0",
  
  schedule: "*/5 * * * *",
  
  async onScheduled() {
    const checks = await Promise.all([
      this.checkCPU(),
      this.checkMemory(),
      this.checkDisk(),
      this.checkServices()
    ]);
    
    for (const check of checks) {
      if (check.healthy === false) {
        await this.handleIssue(check);
      }
    }
  },
  
  async checkCPU() {
    const usage = await this.getCPUUsage();
    return {
      type: 'cpu',
      healthy: usage < 90,
      value: usage,
      threshold: 90
    };
  },
  
  async checkMemory() {
    const usage = await this.getMemoryUsage();
    return {
      type: 'memory',
      healthy: usage < 90,
      value: usage,
      threshold: 90
    };
  },
  
  async checkServices() {
    const services = this.config.checks.services;
    const results = [];
    
    for (const svc of services) {
      const isRunning = await this.isServiceRunning(svc);
      results.push({
        type: 'service',
        name: svc,
        healthy: isRunning,
        service: svc
      });
    }
    
    return results;
  },
  
  async handleIssue(check) {
    const action = this.config.actions[`${check.type}_${check.healthy ? 'high' : 'down'}`];
    
    for (const act of action) {
      if (act === 'restart-service') {
        await this.restartService(check.service);
      } else if (act === 'notify') {
        await this.notifyAdmin(check);
      }
    }
  }
};
```

## 3. PR 代码审查

### 功能
- 自动获取 GitHub PR
- 运行代码分析
- 生成审查意见
- 检查安全漏洞

### 配置

```json
{
  "skills": {
    "code-review": {
      "enabled": true,
      "github": {
        "token": "GITHUB_TOKEN",
        "repos": ["owner/repo1", "owner/repo2"]
      },
      "triggers": ["opened", "synchronize"],
      "checks": ["security", "style", "complexity"]
    }
  }
}
```
