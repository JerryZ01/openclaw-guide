## 场景 1: 企业内部客服机器人

### 需求
- 员工可以通过钉钉/飞书咨询问题
- 自动回复常见问题
- 复杂问题转人工

### 解决方案

```javascript
// skills/enterprise-support/index.js
export default {
  name: "enterprise-support",
  version: "1.0.0",
  
  // 知识库
  knowledgeBase: [
    {
      keywords: ["请假", "休假", "年假"],
      answer: `📝 请假指南：

1. 登录 HR 系统
2. 选择 请假申请
3. 填写开始/结束日期
4. 提交审批

如有疑问请联系 HR 部`
    },
    {
      keywords: ["报销", "费用"],
      answer: `💰 报销流程：

1. 收集发票
2. 登录财务系统
3. 填写报销单
4. 领导审批
5. 财务打款`
    },
    {
      keywords: ["IT", "电脑", "网络"],
      answer: `💻 IT 支持：

- 电脑问题：拨打分机 8001
- 网络问题：拨打分机 8002
- 邮件问题：拨打分机 8003`
    }
  ],
  
  rules: [
    {
      pattern: /.*/,
      handler: async (message) => {
        const content = message.content.toLowerCase();
        
        // 搜索知识库
        for (const item of this.knowledgeBase) {
          if (item.keywords.some(k => content.includes(k))) {
            return item.answer;
          }
        }
        
        // 未匹配，转人工
        return `抱歉，我没有理解您的问题。

您可以：
1. 拨打客服热线 400-xxx-xxxx
2. 发送邮件 support@company.com
3. 继续描述您的问题，我将继续学习`;
      }
    }
  ]
}
```

```json
      "enabled": true,

    }
  }
```

## 场景 2: 订单通知机器人

### 需求
- 订单状态变更时通知用户
- 支持 WhatsApp 和 Telegram

### 解决方案

```javascript
// skills/order-notify/index.js
export default {
  name: "order-notify",
  version: "1.0.0",
  
  // 订单状态模板
  templates: {
    created: `📦 新订单通知

订单号：{{orderId}}
商品：{{product}}
金额：¥{{price}}
状态：已创建`,
    
    paid: `✅ 支付成功

订单号：{{orderId}}
金额：¥{{price}}
支付时间：{{paidAt}}`,
    
    shipped: `🚚 已发货

订单号：{{orderId}}
快递：{{carrier}}
单号：{{trackingNo}}`,
    
    delivered: `✨ 已签收

订单号：{{orderId}}
感谢您的购买！`
  },
  
  // Webhook 处理
  async onWebhook(payload) {
    const { type, orderId, ...data } = payload;
    const template = this.templates[type];
    
    if (!template) return;
    
    // 渲染模板
    let message = template;
    for (const [key, value] of Object.entries(data)) {
      message = message.replace(new RegExp(`{{${key}}}`, 'g'), value);
    }
    
    // 发送通知
    await this.sendToUser(data.userId, message);
  },
  
  async sendToUser(userId, message) {
    // 尝试多个渠道
    const channels = ['whatsapp', 'telegram'];
    
    for (const channel of channels) {
      try {
        await this.sendToChannel(channel, {
          target: userId,
          message: message
        });
        break;
      } catch (e) {
        console.log(`${channel} 失败: ${e.message}`);
      }
    }
  }
}
```

### 集成

```javascript
// 订单系统调用
async function notifyOrderStatus(orderId, status, userInfo) {
  const webhookUrl = process.env.OPENCLAW_WEBHOOK;
  
  await fetch(webhookUrl, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      type: status,
      orderId: orderId,
      ...userInfo
    })
  });
}
```

### 解决方案

```javascript
// skills/daily-report/index.js
export default {
  name: "daily-report",
  schedule: "0 8 * * *", // 每天早上 8 点
  
  async onScheduled() {
    // 获取昨日数据
    const data = await this.gatherData();
    
    // 生成报告
    const report = this.generateReport(data);
    
    // 发送到各渠道
    await this.broadcastReport(report);
  },
  
  async gatherData() {
    // 这里从各系统获取数据
    return {
      orders: { total: 150, revenue: 25000 },
      users: { new: 30, active: 500 },
      tickets: { open: 5, resolved: 20 }
    };
  },
  
  generateReport(data) {
    return `📊 **昨日数据汇总**

**订单**
- 总订单：${data.orders.total}
- 营收：¥${data.orders.revenue}

**用户**
- 新增：${data.users.new}
- 活跃：${data.users.active}

**工单**
- 待处理：${data.tickets.open}
- 已解决：${data.tickets.resolved}

---
数据来源：各业务系统`;
  },
  
  async broadcastReport(report) {
    const targets = [
      { channel: 'telegram', target: 'admin-group' },
      { channel: 'feishu', target: 'ou_xxx' }
    ];
    
    for (const { channel, target } of targets) {
      try {
        await this.sendToChannel(channel, { target, message: report });
      } catch (e) {
        console.error(`${channel} 发送失败:`, e);
      }
    }
  }
}
```

## 场景 4: 智能客服分流

### 需求
- 根据用户问题类型分流到不同机器人
- 简单问题 AI 直接回复
- 复杂问题转人工

### 解决方案

```javascript
// skills/smart-router/index.js
export default {
  name: "smart-router",
  version: "1.0.0",
  
  // 问题分类器
  classifiers = {
    // 技术支持
    technical: ['bug', '错误', '崩溃', '无法', '怎么'],
    // 商务
   商务: ['价格', '报价', '合作', '代理', '合同'],
    // 售后
    afterSales: ['退货', '换货', '维修', '退款', '投诉'],
    // 其他
    default: ['.*']
  },
  
  // 各类型处理器
  handlers: {
    technical: async (message) => {
      // 先尝试知识库
      const answer = await this.searchKB(message.content);
      if (answer) return answer;
      
      // 转技术支持机器人
      return await this.delegateTo('tech-support', message);
    },
    
    商务: async (message) => {
      return await this.delegateTo('sales', message);
    },
    
    afterSales: async (message) => {
      return await this.delegateTo('aftersales', message);
    },
    
    default: async (message) => {
      return await this.ai.chat(message.content);
    }
  },
  
  rules: [
    {
      pattern: /.*/,
      handler: async (message) => {
        const category = this.classify(message.content);
        const handler = this.handlers[category] || this.handlers.default;
        return await handler(message);
      }
    }
  ],
  
  classify(content) {
    for (const [category, keywords] of Object.entries(this.classifiers)) {
      if (keywords.some(k => new RegExp(k).test(content))) {
        return category;
      }
    }
    return 'default';
  },
  
  async searchKB(query) {
    // 搜索知识库
    return null;
  }
}
```

```javascript
  queries: {

          return `未找到查询 "${queryName}"\n\n可用查询：\n${Object.keys(this.queries).join('\n')}`;

        }
        
        const result = await this.executeSQL(message.content);
        return this.formatResult('SQL结果', result);
      }
    }
  ],
  
  async executeSQL(sql) {
    // 连接数据库执行
    return [];
  },
  
  formatResult(title, data) {
    if (!data.length) return `${title}: 无数据`;
    
    const headers = Object.keys(data[0]);
    let result = `📊 ${title}\n\n`;
    
    // 表头
    result += headers.join(' | ') + '\n';
    result += headers.map(() => '---').join(' | ') + '\n';
    
    // 数据行
    for (const row of data.slice(0, 10)) {
      result += headers.map(h => row[h]).join(' | ') + '\n';
    }
    
    if (data.length > 10) {
      result += `\n... 共 ${data.length} 条`;
    }
    
    return result;
  }
```

## 场景 7: 多语言机器人

### 需求
- 自动检测语言
- 多语言回复

### 解决方案

```javascript
// skills/multi-language/index.js
export default {
  name: "multi-language",
  
  // 语言检测
  detectLanguage(text) {
    // 简单检测
    const chinese = /[\u4e00-\u9fa5]/.test(text);
    const japanese = /[\u3040-\u309f\u30a0-\u30ff]/.test(text);
    const korean = /[\uac00-\ud7af]/.test(text);
    
    if (chinese) return 'zh';
    if (japanese) return 'ja';
    if (korean) return 'ko';
    return 'en';
  },
  
  rules: [
    {
      pattern: /.*/,
      handler: async (message) => {
        const lang = this.detectLanguage(message.content);
        
        // 根据语言回复
        const responses = {
          zh: "你好！有什么可以帮助？",
          en: "Hello! How can I help you?",
          ja: "こんにちは！何か手伝いましょうか？",
          ko: "안녕하세요! 무엇을 도와드릴까요?"
        };
        
        // AI 回复（使用对应语言）
        const aiResponse = await this.ai.chat(message.content, {
          language: lang
        });
        
        return aiResponse || responses[lang];
      }
    }
  ]
}
```
