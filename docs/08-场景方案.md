## 场景 2: 订单通知机器人

### 需求
- 订单状态变更时通知用户
- 支持 WhatsApp 和 Telegram

### 解决方案

```javascript
// skills/order-notify/index.js
export default {
  name: "order-notify",
  version: "1.0.0",
  
  // 订单状态模板
  templates: {
    created: `📦 新订单通知

订单号：{{orderId}}
商品：{{product}}
金额：¥{{price}}
状态：已创建`,
    
    paid: `✅ 支付成功

订单号：{{orderId}}
金额：¥{{price}}
支付时间：{{paidAt}}`,
    
    shipped: `🚚 已发货

订单号：{{orderId}}
快递：{{carrier}}
单号：{{trackingNo}}`,
    
    delivered: `✨ 已签收

订单号：{{orderId}}
感谢您的购买！`
  },
  
  // Webhook 处理
  async onWebhook(payload) {
    const { type, orderId, ...data } = payload;
    const template = this.templates[type];
    
    if (!template) return;
    
    // 渲染模板
    let message = template;
    for (const [key, value] of Object.entries(data)) {
      message = message.replace(new RegExp(`{{${key}}}`, 'g'), value);
    }
    
    // 发送通知
    await this.sendToUser(data.userId, message);
  },
  
  async sendToUser(userId, message) {
    // 尝试多个渠道
    const channels = ['whatsapp', 'telegram'];
    
    for (const channel of channels) {
      try {
        await this.sendToChannel(channel, {
          target: userId,
          message: message
        });
        break;
      } catch (e) {
        console.log(`${channel} 失败: ${e.message}`);
      }
    }
  }
}
```

```javascript
// skills/smart-router/index.js
export default {
  name: "smart-router",
  version: "1.0.0",
  
  // 问题分类器
  classifiers = {
    // 技术支持
    technical: ['bug', '错误', '崩溃', '无法', '怎么'],
    // 商务
   商务: ['价格', '报价', '合作', '代理', '合同'],
    // 售后
    afterSales: ['退货', '换货', '维修', '退款', '投诉'],
    // 其他
    default: ['.*']
  },
  
  // 各类型处理器
  handlers: {
    technical: async (message) => {
      // 先尝试知识库
      const answer = await this.searchKB(message.content);
      if (answer) return answer;
      
      // 转技术支持机器人
      return await this.delegateTo('tech-support', message);
    },
    
    商务: async (message) => {
      return await this.delegateTo('sales', message);
    },
    
    afterSales: async (message) => {
      return await this.delegateTo('aftersales', message);
    },
    
    default: async (message) => {
      return await this.ai.chat(message.content);
    }
  },
  
  rules: [
    {
      pattern: /.*/,
      handler: async (message) => {
        const category = this.classify(message.content);
        const handler = this.handlers[category] || this.handlers.default;
        return await handler(message);
      }
    }
  ],
  
  classify(content) {
    for (const [category, keywords] of Object.entries(this.classifiers)) {
      if (keywords.some(k => new RegExp(k).test(content))) {
        return category;
      }
    }
    return 'default';
  },
  
  async searchKB(query) {
    // 搜索知识库
    return null;
  }
}
```

```javascript
// skills/multi-language/index.js
export default {
  name: "multi-language",
  
  // 语言检测
  detectLanguage(text) {
    // 简单检测
    const chinese = /[\u4e00-\u9fa5]/.test(text);
    const japanese = /[\u3040-\u309f\u30a0-\u30ff]/.test(text);
    const korean = /[\uac00-\ud7af]/.test(text);
    
    if (chinese) return 'zh';
    if (japanese) return 'ja';
    if (korean) return 'ko';
    return 'en';
  },
  
  rules: [
    {
      pattern: /.*/,
      handler: async (message) => {
        const lang = this.detectLanguage(message.content);
        
        // 根据语言回复
        const responses = {
          zh: "你好！有什么可以帮助？",
          en: "Hello! How can I help you?",
          ja: "こんにちは！何か手伝いましょうか？",
          ko: "안녕하세요! 무엇을 도와드릴까요?"
        };
        
        // AI 回复（使用对应语言）
        const aiResponse = await this.ai.chat(message.content, {
          language: lang
        });
        
        return aiResponse || responses[lang];
      }
    }
  ]
}
```

```javascript
// skills/home-automation/index.js
export default {
  name: "home-automation",
  
  // 设备控制
  devices: {
    '客厅灯': { type: 'light', protocol: 'mqtt', topic: 'home/livingroom/light' },
    '卧室空调': { type: 'ac', protocol: 'mqtt', topic: 'home/bedroom/ac' },
    '大门锁': { type: 'lock', protocol: 'mqtt', topic: 'home/door/lock' }
  },
  
  // 场景
  scenes: {
    '回家': ['开灯', '开空调', '开门锁'],
    '离家': ['关灯', '关空调', '门锁'],
    '睡觉': ['关灯', '窗帘关闭', '门锁']
  },
  
  rules: [
    {
      pattern: /(开|关)(.+)/,
      handler: async (match) => {
        const action = match[1] === '开' ? 'on' : 'off';
        const device = match[2].trim();
        
        const deviceInfo = this.devices[device];
        if (!deviceInfo) {
          return `未找到设备 "${device}"\n可用设备：${Object.keys(this.devices).join('、')}`;
        }
        
        await this.mqttPublish(deviceInfo.topic, action);
        return `✅ 已${action === 'on' ? '开启' : '关闭'} ${device}`;
      }
    },
    {
      pattern: /执行\s*(.+)/,
      handler: async (match) => {
        const sceneName = match[1];
        const scene = this.scenes[sceneName];
        
        if (!scene) {
          return `未找到场景 "${sceneName}"\n可用场景：${Object.keys(this.scenes).join('、')}`;
        }
        
        for (const action of scene) {
          await this.execute(action);
        }
        
        return `✅ 已执行场景：${sceneName}`;
      }
    },
    {
      pattern: /设备状态/,
      handler: async () => {
        let status = '📊 设备状态\n\n';
        
        for (const [name, device] of Object.entries(this.devices)) {
          const state = await this.mqttSubscribe(device.topic + '/status');
          status += `- ${name}: ${state || '未知'}\n`;
        }
        
        return status;
      }
    }
  ]
}
```

## 场景 9: 早晨简报 (Daily Brief)

### 需求
- 每天早上自动推送天气/日程/新闻摘要
- 定制化内容

### 解决方案

```javascript
// skills/daily-brief/index.js
module.exports = {
  name: "daily-brief",
  version: "1.0.0",
  
  // 定时: 工作日早上 7:30
  schedule: "30 7 * * 1-5",
  
  async onScheduled() {
    const userId = "default_user";
    
    // 获取各项内容
    const weather = await this.getWeather();
    const calendar = await this.getCalendar();
    const tasks = await this.getTasks();
    const news = await this.getNews();
    
    // 生成简报
    const brief = this.generateBrief({ weather, calendar, tasks, news });
    
    // 发送
    await this.sendMessage({
      channel: 'feishu',
      target: userId,
      content: brief
    });
  },
  
  async getWeather() {
    // 调用天气技能
    const result = await this.runSkill('weather', { city: '上海' });
    return result;
  },
  
  async getCalendar() {
    // 获取今日日程
    return [
      { time: '10:00', title: '团队周会', location: '会议室A' },
      { time: '14:00', title: '项目评审', location: '线上' }
    ];
  },
  
  async getTasks() {
    // 获取待办事项
    return [
      { priority: 'high', title: '提交周报' },
      { priority: 'medium', title: '回复邮件' }
    ];
  },
  
  async getNews() {
    // 获取科技新闻 (使用搜索技能)
    const news = await this.webSearch({ query: '科技 今日要闻', limit: 3 });
    return news;
  },
  
  generateBrief({ weather, calendar, tasks, news }) {
    let brief = '☀️ **早上好！**\n\n';
    
    // 天气
    if (weather) {
      const { temp, condition, suggestion } = weather;
      brief += `**今日天气** ${temp}°C ${condition}\n${suggestion}\n\n`;
    }
    
    // 日程
    if (calendar.length) {
      brief += '**📅 今日日程**\n';
      for (const event of calendar) {
        brief += `- ${event.time} ${event.title} (${event.location})\n`;
      }
      brief += '\n';
    }
    
    // 待办
    if (tasks.length) {
      brief += '**✅ 待办事项**\n';
      for (const task of tasks) {
        const emoji = task.priority === 'high' ? '🔴' : '🟡';
        brief += `${emoji} ${task.title}\n`;
      }
      brief += '\n';
    }
    
    // 新闻
    if (news.length) {
      brief += '**📰 科技要闻**\n';
      for (const item of news.slice(0, 3)) {
        brief += `- ${item.title}\n`;
      }
    }
    
    brief += '\n---';
    brief += '\n💡 美好的一天从此刻开始！';
    
    return brief;
  }
};
```

## 场景 10: 安全远程 SSH

### 需求
- 通过 Telegram 安全远程执行 SSH 命令
- 避免公网暴露 SSH 端口

### 解决方案

```javascript
// skills/homelab-access/index.js
module.exports = {
  name: "homelab-access",
  version: "1.0.0",
  
  // 授权用户
  authorizedUsers: ['telegram_user_id_1', 'telegram_user_id_2'],
  
  // SSH 配置
  servers: {
    'nas': {
      host: '192.168.1.100',
      port: 22,
      user: 'admin',
      keyPath: '~/.ssh/id_rsa'
    },
    '服务器': {
      host: '192.168.1.200',
      port: 22,
      user: 'deploy'
    }
  },
  
  // 预定义命令
  presets: {
    '状态': 'uptime && df -h',
    '内存': 'free -h',
    '进程': 'ps aux | head -20',
    '日志': 'tail -50 /var/log/syslog'
  },
  
  rules: [
    {
      pattern: /^ssh\s+(\w+)\s+(.+)/,
      handler: async (match, context) => {
        const serverName = match[1];
        const command = match[2];
        
        // 权限检查
        if (!this.authorizedUsers.includes(context.from)) {
          return '❌ 未授权';
        }
        
        const server = this.servers[serverName];
        if (!server) {
          return `未找到服务器 "${serverName}"\n可用: ${Object.keys(this.servers).join(', ')}`;
        }
        
        const result = await this.sshExec(server, command);
        return `\`${serverName}$\` ${command}\n\`\`\`\n${result}\n\`\`\``;
      }
    },
    {
      pattern: /^预置\s+(\w+)\s+(.+)/,
      handler: async (match, context) => {
        const preset = match[1];
        const serverName = match[2];
        
        if (!this.presets[preset]) {
          return `未找到预置命令 "${preset}"\n可用: ${Object.keys(this.presets).join(', ')}`;
        }
        
        const server = this.servers[serverName];
        if (!server) {
          return `未找到服务器 "${serverName}"`;
        }
        
        const result = await this.sshExec(server, this.presets[preset]);
        return `\`${serverName}$\` ${this.presets[preset]}\n\`\`\`\n${result}\n\`\`\``;
      }
    },
    {
      pattern: /^服务器列表/,
      handler: async () => {
        let list = '🖥️ 可用服务器\n\n';
        for (const [name, server] of Object.entries(this.servers)) {
          list += `- **${name}**: ${server.user}@${server.host}:${server.port}\n`;
        }
        return list;
      }
    }
  ],
  
  async sshExec(server, command) {
    // 使用 ssh2 库执行命令
    const { Client } = require('ssh2');
    
    return new Promise((resolve, reject) => {
      const conn = new Client();
      conn.on('ready', () => {
        conn.exec(command, (err, stream) => {
          if (err) {
            conn.end();
            return reject(err);
          }
          
          let output = '';
          stream.on('close', () => {
            conn.end();
            resolve(output);
          }).on('data', (data) => {
            output += data.toString();
          }).stderr.on('data', (data) => {
            output += data.toString();
          });
        });
      }).connect({
        host: server.host,
        port: server.port,
        username: server.user,
        privateKey: require('fs').readFileSync(require('path').expand(server.keyPath))
      });
    });
  }
};
```

```javascript
// skills/tech-discoveries/index.js
module.exports = {
  name: "tech-discoveries",
  version: "1.0.0",
  
  schedule: "0 9 * * 1,3,5", // 周一、三、五早上9点
  
  // 新闻源
  sources: [
    { name: 'Hacker News', url: 'https://news.ycombinator.com', selector: '.athing' },
    { name: 'TechCrunch', url: 'https://techcrunch.com', selector: 'article' },
    { name: 'InfoQ', url: 'https://infoq.com', selector: '.news_summary' }
  ],
  
  // 关键词过滤
  interests: ['openclaw', 'ai', 'llm', 'automation', 'agent'],
  
  async onScheduled() {
    const articles = await this.gatherNews();
    const filtered = this.filter(articles);
    const summarized = await this.summarize(filtered);
    
    const report = this.formatReport(summarized);
    
    await this.broadcast(report);
  },
  
  async gatherNews() {
    const articles = [];
    
    for (const source of this.sources) {
      try {
        const html = await this.fetch(source.url);
        const parsed = this.parse(html, source.selector);
        articles.push(...parsed.map(a => ({ ...a, source: source.name })));
      } catch (e) {
        console.error(`获取 ${source.name} 失败:`, e.message);
      }
    }
    
    return articles;
  },
  
  filter(articles) {
    return articles.filter(article => {
      const text = (article.title + article.description).toLowerCase();
      return this.interests.some(keyword => text.includes(keyword));
    });
  },
  
  async summarize(articles) {
    return articles.slice(0, 5).map(article => ({
      title: article.title,
      source: article.source,
      url: article.url,
      // 使用 AI 摘要
      summary: article.description?.substring(0, 200)
    }));
  },
  
  formatReport(articles) {
    let report = '📰 **技术要闻**\n';
    report += `> 精选 ${this.interests.join(', ')} 相关\n\n`;
    
    for (const article of articles) {
      report += `**${article.source}**: [${article.title}](${article.url})\n`;
      if (article.summary) {
        report += `> ${article.summary}...\n`;
      }
      report += '\n';
    }
    
    return report;
  },
  
  async broadcast(report) {
    await this.sendToChannel('telegram', { target: 'tech-group', message: report });
    await this.sendToChannel('feishu', { target: 'news-channel', message: report });
  }
};
```
